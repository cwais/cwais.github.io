CREATE TABLE tmp_report_24672_camp_with_dimcat (PRIMARY KEY(campaign_id_in_kenshoo)) AS             SELECT                               CAST(IFNULL(GROUP_CONCAT(IF(INSTR(dimcat.category_name,'674,')=1,dimcat.category_name,null) SEPARATOR '|'),'674,-1') as CHAR) as categoryName_674,CAST(IFNULL(GROUP_CONCAT(IF(INSTR(dimcat.category_name,'663,')=1,dimcat.category_name,null) SEPARATOR '|'),'663,-1') as CHAR) as categoryName_663,CAST(IFNULL(GROUP_CONCAT(IF(INSTR(dimcat.category_name,'494,')=1,dimcat.category_name,null) SEPARATOR '|'),'494,-1') as CHAR) as categoryName_494,CAST(IFNULL(GROUP_CONCAT(IF(INSTR(dimcat.category_name,'428,')=1,dimcat.category_name,null) SEPARATOR '|'),'428,-1') as CHAR) as categoryName_428,                            sc.*   FROM tmp_report_24672_struct_campaigns sc       LEFT JOIN tmp_report_24672_dimcat dimcat ON dimcat.campaign_id = sc.campaign_id_in_kenshoo     GROUP BY sc.campaign_id_in_kenshoo
